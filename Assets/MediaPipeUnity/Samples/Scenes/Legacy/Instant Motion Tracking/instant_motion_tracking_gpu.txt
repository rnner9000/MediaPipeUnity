# MediaPipe graph that performs region tracking and visualizes tracking results.
# GPU version.

# Input GPU image (GpuBuffer)
input_stream: "input_video"

# Output GPU image with region tracking visualization (GpuBuffer)
output_stream: "output_video"

# Flow limiter to maintain proper pipeline throughput
node {
  calculator: "FlowLimiterCalculator"
  input_stream: "input_video"
  input_stream: "FINISHED:output_video"
  input_stream_info: {
    tag_index: "FINISHED"
    back_edge: true
  }
  output_stream: "throttled_input_video"
}

# Region tracking subgraph - detects and tracks rectangular regions
node {
  calculator: "RegionTrackingSubgraph"
  input_stream: "VIDEO:throttled_input_video"
  output_stream: "ANCHORS:tracked_regions"
}

# GPU-based visualization subgraph - renders tracked regions on the image
node {
  calculator: "GlAnnotationOverlayCalculator"
  input_stream: "IMAGE:throttled_input_video"
  input_stream: "ANCHORS:tracked_regions"
  output_stream: "IMAGE:output_video_raw"
}

# Apply output GPU transformations
node: {
  calculator: "GlScalerCalculator"
  input_stream: "VIDEO:output_video_raw"
  output_stream: "VIDEO:output_video"
}
